Start it up:

#+begin_src bash
  docker volume create n8n_data

  docker run -it --rm \
         --add-host=host.docker.internal:host-gateway \
         --name n8n -p 5678:5678 -v n8n_data:/home/node/.n8n \
         docker.n8n.io/n8nio/n8n
#+end_src
